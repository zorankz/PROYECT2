  <script>
    function mostrarReloj() {
      const numero = document.getElementById('numeroA').value.trim();
      if (!/^\d{9}$/.test(numero)) {
        alert('Por favor ingresa un número A válido de 9 dígitos.');
        return;
      }
      const iframe = document.getElementById('iframeReloj');
      iframe.src = `https://migraconnect.us/eoir-tracker/${numero}`;
      document.getElementById('resultado').style.display = 'block';
    }
